\begin{frame}{Актуальность}
  \footnotesize
  \begin{itemize}
    \setlength{\itemsep}{0.4em}
    \item Криптография --- повседневная: TLS, мессенджеры, банкинг, IoT, аппаратные кошельки $\Rightarrow$ миллиарды ключей в год.
    \item Массовые микроконтроллеры (Arduino, ESP32, AVR, STM32 базовых серий) не имеют аппаратного TRNG.
    \item Современные стандарты (NIST SP 800-90B, ГОСТ Р 34.10) требуют физического источника энтропии для ключевого материала.
    \item Цена ошибки: \textbf{Android 2012} --- ${\approx}\,86\%$ Bitcoin-ключей предсказуемы; \textbf{ROCA} --- уязвимы миллионы сертифицированных TPM.
  \end{itemize}
  \vspace{0.4em}
\end{frame}

% ============================================================
\begin{frame}{Цель и задачи}
  \textbf{Цель.} Сравнить физические источники энтропии и спроектировать компактный аппаратный TRNG.
  \vspace{0.6em}
  \textbf{Задачи:}
  \begin{itemize}
    \item теоретический анализ восьми классов источников;
    \item сборка стенда на доступной элементной базе;
    \item практическая проверка шести источников;
    \item оценка по NIST STS, скорости, стоимости.
  \end{itemize}
\end{frame}
